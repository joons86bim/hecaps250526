// 현재 URN들 확인
const RAW = window.CURRENT_MODEL_URN;
const SAFE = window.CURRENT_MODEL_SAFE_URN;
console.log({ RAW, SAFE });

// 저장소에 RAW/SAFE 둘 다 있는지 확인
const kRaw  = `hec:tasks:${RAW}`;
const kSafe = `hec:tasks:${SAFE}`;
const rawJson  = localStorage.getItem(kRaw);
const safeJson = localStorage.getItem(kSafe);
console.log("[LS] keys", { kRaw, hasRaw: !!rawJson, kSafe, hasSafe: !!safeJson });

// 메모리 스냅샷도 확인
